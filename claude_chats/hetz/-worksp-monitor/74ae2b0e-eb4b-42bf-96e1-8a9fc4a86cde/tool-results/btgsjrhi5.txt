const WINDOW_MS = 60_000;

function checkRateLimit(ip: string): boolean {
  const now = Date.now();
  const entry = rateLimiter.get(ip);

  if (!entry || now > entry.resetAt) {

  return true;
}

  const expected = process.env.ISSUE_WORKER_TOKEN;

  return token === expected;
