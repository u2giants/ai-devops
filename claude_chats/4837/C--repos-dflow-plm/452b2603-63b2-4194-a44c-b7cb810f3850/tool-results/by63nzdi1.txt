    10	// Helper function to format dates in user-friendly format
    11	function formatUserFriendlyDate(dateValue) {
    12	  if (!dateValue) return null;
    13	  
    14	  try {
    15	    let date = new Date(dateValue);
    16	    // Fix timezone shift for YYYY-MM-DD strings
    17	    if (typeof dateValue === 'string' && dateValue.includes('-') && !dateValue.includes('T')) {
    18	      const [y, m, day] = dateValue.split('-');
    19	      date = new Date(Number(y), Number(m) - 1, Number(day));
    20	    }
    21	    if (isNaN(date.getTime())) return null;
    22	    
    23	    // Format as MM/DD/YYYY
    24	    return date.toLocaleDateString('en-US', {
    25	      year: 'numeric',
    26	      month: '2-digit',
    27	      day: '2-digit'
    28	    });
    29	  } catch (error) {
    30	    return null;
    31	  }
    32	}

    34	// Format dates as ordinal: "3rd Dec 25" for stage tracker display
    35	function formatOrdinalDate(dateValue) {
    36	  if (!dateValue) return null;
    37	  try {
    38	    let d = new Date(dateValue);
    39	    // Fix timezone shift for YYYY-MM-DD strings
    40	    if (typeof dateValue === 'string' && dateValue.includes('-') && !dateValue.includes('T')) {
    41	      const [y, m, day] = dateValue.split('-');
    42	      d = new Date(Number(y), Number(m) - 1, Number(day));
    43	    }
    44	    if (isNaN(d.getTime())) return null;
    45	    const day = d.getDate();
    46	    const suffix = [,'st','nd','rd'][day % 10 > 3 ? 0 : (day % 100 - day % 10 !== 10) * (day % 10)] || 'th';
    47	    const month = d.toLocaleString('en-US', { month: 'short' });
    48	    const year = String(d.getFullYear()).slice(-2);
    49	    return `${day}${suffix} ${month} ${year}`;
    50	  } catch (error) {
    51	    return null;
    52	  }
    53	}

    55	function stripHtml(html) {
    56	  if (!html) return html;
    57	  let text = html.replace(/<[^>]*>?/gm, '');
    58	  const entities = {
    59	    '&nbsp;': ' ',
    60	    '&lt;': '<',
    61	    '&gt;': '>',
    62	    '&amp;': '&',
    63	    '&quot;': '"',
    64	    '&apos;': "'"
    65	  };
    66	  return text.replace(/&[a-z]+;/g, match => entities[match] || match);
    67	}

    93	/**
    94	 * Build 2-3 fixed licensing stage rows from raw licensingStatus DB entries.
    95	 * Stages: Submission, Resubmission (if revisions happened), PPS Approval (if PPS entries exist).
    96	 * licTimes: { submission_days, resubmission_days, pps_approval_days } or null.
    97	 * level3NeededDate: Date|string — the needed-by date for licensing to be complete (from caller).
    98	 */
    99	function buildStageRows(statuses, licTimes, level3NeededDate, milestones = {}) {
   100	  const hasLinesheetData = milestones.linesheet_submission || milestones.linesheet_resubmission;
   101	  if ((!statuses || !statuses.length) && !hasLinesheetData) return [];
   102	
   103	  // Strip time from a date before comparing so "11 PM" on deadline day isn't flagged as delayed
   104	  const toDay = (dateStr) => {
   105	    const d = new Date(dateStr);
   106	    if (isNaN(d.getTime())) return null;
   107	    return new Date(d.getFullYear(), d.getMonth(), d.getDate());
   108	  };
   109	  const isAfterDay = (actual, estimate) => {
   110	    const a = toDay(actual);
   111	    const e = toDay(estimate);
   112	    return !!(a && e && a > e);
   113	  };
   114	
   115	  const subDays = (licTimes && Number(licTimes.submission_days)) || 0;
   116	  const resubDays = (licTimes && Number(licTimes.resubmission_days)) || 0;
   117	  const ppsDays = (licTimes && Number(licTimes.pps_approval_days)) || 0;
   118	
   119	  // Sort all statuses chronologically for reliable sequential lookups
   120	  const sorted = [...statuses].sort((a, b) => new Date(a.date) - new Date(b.date));
   121	
   122	  // Anchor dates
   123	  let conceptSubmittedEntry = sorted.find(s =>
   124	    s.status === 'Concept Submitted' || s.status === 'Packaging Concept Submitted'
   125	  );
   126	  if (!conceptSubmittedEntry && sorted.length > 0) {
   127	    const firstStatus = sorted[0].status || '';
   128	    if (!firstStatus.includes('Pre-pro') && !PPS_STATUSES.has(firstStatus)) {
   129	      conceptSubmittedEntry = sorted[0];
   130	    }
   131	  }
   132	  const conceptSubmittedDate = conceptSubmittedEntry ? conceptSubmittedEntry.date : null;
   133	
   134	  // If a resubmission exists, use the LATEST resubmission as the PPS anchor
   135	  // so that est dates recalculate from the revision, not the original submission
   136	  const resubEntries = sorted.filter(s => s.status === 'Pre-Pro Resubmitted');
   137	  let ppsAnchorEntry = resubEntries.length > 0
   138	    ? resubEntries[resubEntries.length - 1]  // latest resubmission
   139	    : (sorted.find(s => s.status === 'Request Pre-Pro Sample')
   140	      || sorted.find(s => s.status === 'Pre-Pro Submitted')
   141	      || sorted.find(s => s.status === 'Pre-pro Photos Received'));
   142	  if (!ppsAnchorEntry && sorted.length > 0) {
   143	    ppsAnchorEntry = sorted.find(s => PPS_STATUSES.has(s.status) || (s.status || '').includes('Pre-pro'));
   144	  }
   145	  const ppsStartDate = ppsAnchorEntry ? ppsAnchorEntry.date : null;
   146	
   147	  const hasRevisions = sorted.some(s =>
   148	    s.status === 'Concept Needs Revisions' || s.status === 'Packaging Concept Needs Revisions' ||
   149	    s.status === 'Concept Not Approved at All'
   150	  );
   151	  const hasPPS = sorted.some(s => PPS_STATUSES.has(s.status));
   152	
   153	  // ---- Needed-date back-calculation from level3NeededDate ----
   154	  // PPS needed = level3NeededDate (last step must finish by this date)
   155	  // Resubmission needed = PPS needed − pps_approval_days − 1 (1-day buffer between stages)
   156	  // Submission needed = Resubmission needed − resubmission_days − 1
   157	  const ppsNeededDate = level3NeededDate ? formatDate(new Date(level3NeededDate)) : null;
   158	  const resubNeededDate = (level3NeededDate && ppsDays)
   159	    ? formatDate(addDays(level3NeededDate, -(ppsDays + 1)))
   160	    : (level3NeededDate ? formatDate(new Date(level3NeededDate)) : null);
   161	  const subNeededDate = (resubNeededDate && resubDays)
   162	    ? formatDate(addDays(resubNeededDate, -(resubDays + 1)))
   163	    : (resubNeededDate || null);
   164	
   165	  const rows = [];
   166	
   167	  // --- Stage 0a: Linesheet Submission (from milestone data only) ---
   168	  if (milestones.linesheet_submission) {
   169	    const lsDate = milestones.linesheet_submission;
   170	    rows.push({
   171	      phase: 'Linesheet Submission',
   172	      status: 'Submitted',
   173	      date: formatOrdinalDate(lsDate),
   174	      completionDate: formatOrdinalDate(lsDate),
   175	      _rawCompletionDate: lsDate,
   176	      estStartDate: formatOrdinalDate(lsDate),
   177	      estCompletionDate: formatOrdinalDate(lsDate),
   178	      _rawEstCompletionDate: lsDate,
   179	      neededDate: null,
   180	      isDelayed: false
   181	    });
   182	  }
   183	
   184	  // --- Stage 0b: Linesheet Re-submission (from milestone data only) ---
   185	  if (milestones.linesheet_resubmission) {
   186	    const lrDate = milestones.linesheet_resubmission;
   187	    rows.push({
   188	      phase: 'Linesheet Re-submission',
   189	      status: 'Submitted',
   190	      date: formatOrdinalDate(lrDate),
   191	      completionDate: formatOrdinalDate(lrDate),
   192	      _rawCompletionDate: lrDate,
   193	      estStartDate: formatOrdinalDate(lrDate),
   194	      estCompletionDate: formatOrdinalDate(lrDate),
   195	      _rawEstCompletionDate: lrDate,
   196	      neededDate: null,
   197	      isDelayed: false
   198	    });
   199	  }
   200	
   201	  // Early return if there are no licensing statuses — linesheet rows are all we have
   202	  if (!statuses || !statuses.length) return rows;
   203	
   204	  // --- Stage 1: Submission ---
   205	  if (conceptSubmittedDate) {
   206	    const estComp = formatDate(addDays(conceptSubmittedDate, subDays));
   207	    const terminalEntry = sorted.find(s => SUBMISSION_TERMINAL_STATUSES.has(s.status));
   208	    const completionDate = terminalEntry ? terminalEntry.date : null;
   209	    const statusLabel = terminalEntry
   210	      ? (SUBMISSION_STATUS_LABEL[terminalEntry.status] || terminalEntry.status)
   211	      : 'In Progress';
   212	    // Delayed = completed after deadline, OR deadline passed and still not complete
   213	    const isDelayed = subNeededDate
   214	      ? (completionDate ? isAfterDay(completionDate, subNeededDate) : isAfterDay(new Date(), subNeededDate))
   215	      : false;
   216	    rows.push({
   217	      phase: 'Submission',
   218	      status: statusLabel,
   219	      date: formatOrdinalDate(conceptSubmittedDate),
   220	      completionDate: formatOrdinalDate(completionDate),
   221	      _rawCompletionDate: completionDate,
   222	      estStartDate: formatOrdinalDate(conceptSubmittedDate),
   223	      estCompletionDate: formatOrdinalDate(estComp),
   224	      _rawEstCompletionDate: estComp,
   225	      neededDate: formatOrdinalDate(subNeededDate),
   226	      neededDate_tooltip: subNeededDate
   227	        ? `Needed = Resubmission needed \u2212 resubmission days (${resubDays}d) \u2212 1d buffer\nSubmission days: ${subDays}d\n${licTimes ? `Licensor: ${licTimes.licensor_name || 'Unknown'}` : ''}`
   228	        : null,
   229	      isDelayed
   230	    });
   231	  }
   232	
   233	  // --- Stage 2: Resubmission (only if revisions occurred) ---
   234	  if (conceptSubmittedDate && hasRevisions) {
   235	    // estStart chains from previous stage's actual completion + 1 day (buffer),
   236	    // or previous stage's estCompletion + 1 day if no actual completion yet.
   237	    const submissionRow = rows[rows.length - 1];
   238	    const prevRawCompletion = submissionRow ? submissionRow._rawCompletionDate : null;
   239	    const prevRawEstComp = submissionRow ? submissionRow._rawEstCompletionDate : null;
   240	    const resubEstStart = prevRawCompletion
   241	      ? formatDate(addDays(prevRawCompletion, 1))
   242	      : formatDate(addDays(prevRawEstComp, 1));
   243	    const estComp = formatDate(addDays(resubEstStart, resubDays));
   244	
   245	    const resubEntry = sorted.find(s =>
   246	      s.status === 'Concept Resubmitted' || s.status === 'Packaging Concept Resubmitted'
   247	    );
   248	    const resubDate = resubEntry ? resubEntry.date : null;
   249	
   250	    // Find the first "Concept Approved" that came after the resubmission
   251	    let resubApprovalDate = null;
   252	    if (resubEntry) {
   253	      const afterResub = sorted.filter(s => new Date(s.date) > new Date(resubEntry.date));
   254	      const approvedAfter = afterResub.find(s =>
   255	        s.status === 'Concept Approved' || s.status === 'Concept approved with Changes' ||
   256	        s.status === 'Packaging Concept Approved' || s.status === 'Packaging Concept Approved with Changes'
   257	      );
   258	      resubApprovalDate = approvedAfter ? approvedAfter.date : null;
   259	    }
   260	
   261	    let statusLabel = 'In Progress';
   262	    if (resubApprovalDate) statusLabel = 'Approved';
   263	    else if (resubDate) statusLabel = 'Resubmitted';
   264	
   265	    // Delayed = completed after deadline, OR deadline passed and still not complete
   266	    const isDelayed = resubNeededDate
   267	      ? (resubApprovalDate ? isAfterDay(resubApprovalDate, resubNeededDate) : isAfterDay(new Date(), resubNeededDate))
   268	      : false;
   269	    rows.push({
   270	      phase: 'Resubmission',
   271	      status: statusLabel,
   272	      date: formatOrdinalDate(resubDate),
   273	      completionDate: formatOrdinalDate(resubApprovalDate),
   274	      _rawCompletionDate: resubApprovalDate,
   275	      estStartDate: formatOrdinalDate(resubEstStart),
   276	      estCompletionDate: formatOrdinalDate(estComp),
   277	      _rawEstCompletionDate: estComp,
   278	      neededDate: formatOrdinalDate(resubNeededDate),
   279	      neededDate_tooltip: resubNeededDate
   280	        ? `Needed = PPS needed \u2212 PPS approval days (${ppsDays}d) \u2212 1d buffer\nResubmission days: ${resubDays}d\n${licTimes ? `Licensor: ${licTimes.licensor_name || 'Unknown'}` : ''}`
   281	        : null,
   282	      isDelayed
   283	    });
   284	  }
   285	
   286	  // --- Stage 3: PPS Approval (only if PPS statuses exist) ---
   287	  if (hasPPS) {
   288	    // Prefer the DB anchor date; fall back to chaining from the last stage + 1 day buffer
   289	    const lastRow = rows.length ? rows[rows.length - 1] : null;
   290	    const ppsEstStart = ppsStartDate
   291	      ? ppsStartDate
   292	      : (lastRow && lastRow._rawCompletionDate)
   293	        ? formatDate(addDays(lastRow._rawCompletionDate, 1))
   294	        : (lastRow && lastRow._rawEstCompletionDate)
   295	          ? formatDate(addDays(lastRow._rawEstCompletionDate, 1))
   296	          : null;
   297	
   298	    if (ppsEstStart) {
   299	      const estComp = formatDate(addDays(ppsEstStart, ppsDays));
   300	      const ppsApprovedEntry = sorted.find(s =>
   301	        s.status === 'Pre-pro Approved' || s.status === 'Pre-pro approved with Changes'
   302	      );
   303	      const ppsCompletionDate = ppsApprovedEntry ? ppsApprovedEntry.date : null;
   304	
   305	      let statusLabel = 'In Progress';
   306	      if (ppsCompletionDate) {
   307	        statusLabel = ppsApprovedEntry.status === 'Pre-pro approved with Changes'
   308	          ? 'Approved with Changes'
   309	          : 'Approved';
   310	      } else if (sorted.some(s => s.status === 'Pre-pro Rejected')) {
   311	        statusLabel = 'Rejected - Resubmit';
   312	      } else if (sorted.some(s => s.status === 'Pre-pro Needs Revisions')) {
   313	        statusLabel = 'Needs Revisions';
   314	      }
   315	
   316	      // Delayed = completed after deadline, OR deadline passed and still not complete
   317	      const isDelayed = ppsNeededDate
   318	        ? (ppsCompletionDate ? isAfterDay(ppsCompletionDate, ppsNeededDate) : isAfterDay(new Date(), ppsNeededDate))
   319	        : false;
   320	      rows.push({
   321	        phase: 'PPS Approval',
   322	        status: statusLabel,
   323	        date: formatOrdinalDate(ppsStartDate || ppsEstStart),
   324	        completionDate: formatOrdinalDate(ppsCompletionDate),
   325	        _rawCompletionDate: ppsCompletionDate,
   326	        estStartDate: formatOrdinalDate(ppsEstStart),
   327	        estCompletionDate: formatOrdinalDate(estComp),
   328	        _rawEstCompletionDate: estComp,
   329	        neededDate: formatOrdinalDate(ppsNeededDate),
   330	        neededDate_tooltip: ppsNeededDate
   331	          ? `Needed = Level 3 needed date (licensing complete by)\nPPS approval days: ${ppsDays}d\n${licTimes ? `Licensor: ${licTimes.licensor_name || 'Unknown'}` : ''}`
   332	          : null,
   333	        isDelayed
   334	      });
   335	    }
   336	  }
   337	
   338	  return rows;
   339	}

   341	/**
   342	 * Build ALL historical licensing stage rows (Level 4 grid only).
   343	 * Maps every raw status attempt chronologically, hiding estimates/needed dates 
   344	 * for past failures to keep the grid uncluttered, but showing them for the active/latest phase.
   345	 */
   346	function buildHistoricalStageRows(statuses, licTimes, level3NeededDate, milestones = {}) {
   347	  const canonical = buildStageRows(statuses, licTimes, level3NeededDate, milestones);
   348	  const sorted = [...(statuses || [])].sort((a, b) => new Date(a.date) - new Date(b.date));
   349	  
   350	  if (!sorted.length) return canonical; // linesheet fallback
   351	  
   352	  // Find the LAST index of each phase bucket
   353	  let lastSubIdx = -1, lastResubIdx = -1, lastPpsIdx = -1;
   354	  const bucketMap = new Array(sorted.length).fill(null);
   355	  
   356	  for (let i = 0; i < sorted.length; i++) {
   357	    const st = sorted[i].status || '';
   358	    if (PPS_STATUSES.has(st) || st.includes('Pre-pro')) {
   359	      bucketMap[i] = 'PPS Approval';
   360	      lastPpsIdx = i;
   361	    } else if (st.includes('Resubmitted') || st === 'Concept approved with Changes') {
   362	      bucketMap[i] = 'Resubmission';
   363	      lastResubIdx = i;
   364	    } else {
   365	      bucketMap[i] = 'Submission';
   366	      lastSubIdx = i;
   367	    }
   368	  }
   369	
   370	  const historyRows = [];
   371	  for (let i = 0; i < sorted.length; i++) {
   372	    const s = sorted[i];
   373	    
   374	    // Default historical values (blank)
   375	    let estStart = '—';
   376	    let estComp = '—';
   377	    let needed = '—';
   378	    let tooltip = null;
   379	    
   380	    // Re-surface the computed canonical dates onto the final historical row of its logical phase
   381	    let isPhaseFinal = false;
   382	    let canonMatch = null;
   383	    
   384	    if (i === lastPpsIdx) {
   385	      isPhaseFinal = true;
   386	      canonMatch = canonical.find(c => c.phase === 'PPS Approval');
   387	    } else if (i === lastResubIdx) {
   388	      isPhaseFinal = true;
   389	      canonMatch = canonical.find(c => c.phase === 'Resubmission');
   390	    } else if (i === lastSubIdx) {
   391	      isPhaseFinal = true;
   392	      canonMatch = canonical.find(c => c.phase === 'Submission');
   393	    }
   394	
   395	    if (isPhaseFinal && canonMatch) {
   396	      estStart = canonMatch.estStartDate || '—';
   397	      estComp = canonMatch.estCompletionDate || '—';
   398	      needed = canonMatch.neededDate || '—';
   399	      tooltip = canonMatch.neededDate_tooltip;
   400	    }
   401	    
   402	    historyRows.push({
   403	      phase: s.status, // Use literal event name so history is transparent
   404	      status: SUBMISSION_STATUS_LABEL[s.status] || s.status,
   405	      date: formatOrdinalDate(s.date),
   406	      completionDate: formatOrdinalDate(s.date),
   407	      _rawCompletionDate: s.date,
   408	      estStartDate: estStart,
   409	      estCompletionDate: estComp,
   410	      neededDate: needed,
   411	      neededDate_tooltip: tooltip,
   412	      isDelayed: false,
   413	      isHistorical: (estStart === '—' && estComp === '—' && needed === '—')
   414	    });
   415	  }
   416	  
   417	  return historyRows;
   418	}

   420	/**
   421	 * Build the PO-level production timeline (12 stages: 3 sampling + 9 production).
   422	 * This runs at the order level, NOT per-SKU.
   423	 *
   424	 * Sampling stages pull dates from licensing statuses (Tech Pack Sent, Photos Received)
   425	 * and the new sample_start_date field on itemHeader.
   426	 *
   427	 * @param {Object} headerData - ProdOrderHeader dataValues
   428	 * @param {Array} details - Remapped detail rows (with subItems containing licensing data)
   429	 * @returns {Array} Array of stage objects for the production stage tracker
   430	 */
   431	function buildProductionStages(headerData, details) {
   432	  const toDay = (dateStr) => {
   433	    if (!dateStr) return null;
   434	    const d = new Date(dateStr);
   435	    return isNaN(d.getTime()) ? null : new Date(d.getFullYear(), d.getMonth(), d.getDate());
   436	  };
   437	  const isAfterDay = (dateA, dateB) => {
   438	    const a = toDay(dateA);
   439	    const b = toDay(dateB);
   440	    return !!(a && b && a > b);
   441	  };
   442	  // Status based on needed date (deadline), not estimated date
   443	  const stageStatus = (actualDate, neededDate) => {
   444	    if (actualDate) return isAfterDay(actualDate, neededDate) ? 'Delayed' : 'Complete';
   445	    if (neededDate && toDay(neededDate) && toDay(neededDate) < new Date()) return 'Overdue';
   446	    return 'Pending';
   447	  };
   448	
   449	  const massProdDays = Number(headerData.massProductionDays) || 35; // default 35 days
   450	  const FACTORY_PREP_BUFFER = 7; // 1 week buffer after PPS approval
   451	
   452	  // ---- Back-calculate needed dates from Ship Date ----
   453	  // Each stage's deadline works backwards from shipping
   454	  const shipDate = headerData.shipDate ? toDay(headerData.shipDate) : null;
   455	  const QC_BUFFER = 3;
   456	  const PACKING_BUFFER = 3;
   457	  const MASS_PROD_COMPLETE_BUFFER = 1;
   458	  const ASSEMBLY_BUFFER = 7;
   459	  const MASS_PROD_START_BUFFER = 1;
   460	  const MATERIAL_BUFFER = 1;
   461	  const FACTORY_PREP_NEEDED_BUFFER = 7;
   462	  const PPS_BUFFER = 1;
   463	
   464	  const neededShipping      = shipDate ? formatDate(shipDate) : null;
   465	  const neededQC            = shipDate ? formatDate(addDays(shipDate, -QC_BUFFER)) : null;
   466	  const neededPacking       = neededQC ? formatDate(addDays(neededQC, -PACKING_BUFFER)) : null;
   467	  const neededProdComplete  = neededPacking ? formatDate(addDays(neededPacking, -MASS_PROD_COMPLETE_BUFFER)) : null;
   468	  // Mass Prod Start must happen 'massProdDays' before Prod Complete
   469	  const neededProdStart     = neededProdComplete ? formatDate(addDays(neededProdComplete, -massProdDays)) : null;
   470	  // Assembly happens inside the mass production block (e.g. 7 days before complete)
   471	  const neededAssembly      = neededProdComplete ? formatDate(addDays(neededProdComplete, -ASSEMBLY_BUFFER)) : null;
   472	  const neededMaterial      = neededProdStart ? formatDate(addDays(neededProdStart, -MATERIAL_BUFFER)) : null;
   473	  const neededFactoryPrep   = neededMaterial ? formatDate(addDays(neededMaterial, -FACTORY_PREP_NEEDED_BUFFER)) : null;
   474	  const neededPPS           = neededFactoryPrep ? formatDate(addDays(neededFactoryPrep, -PPS_BUFFER)) : null;
   475	
   476	  // ---- Determine if PO is licensed or non-licensed ----
   477	  // Non-licensed div_code_fk values: 2 (Gen/Everyday), 9 (Spruce non-Lic)
   478	  const NON_LICENSED_DIV_CODES = new Set([2, 9]);
   479	  const allSubItems = details.flatMap(d => d.subItems || []);
