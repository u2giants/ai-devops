     6	async function initialize() {
     7	    // connect to db
     8	    let sq_config = {
     9	        dialect: 'postgres',
    10	        host: DBConfig.HOST,
    11	        schema: process.env.SCHEMA,
    12	        logging: false,
    13	        pool: {
    14	            max: 22,
    15	            min: 1,
    16	            acquire: 600000,
    17	            idle: 100000,
    18	        },
    19	        createdAt: false,
    20	        updatedAt: false,
    21	        timestamp: false,
    22	        dialectOptions: {
    23	            connectTimeout: 3000000,
    24	        }
    25	    };
    26	
    27	    const sequelize = new Sequelize(DBConfig.DB, DBConfig.DBUser, DBConfig.PASSWORD, sq_config);
    28	
    29	    // Populate the exported `db` object SYNCHRONOUSLY (before any await). Model
    30	    // definitions don't need an authenticated connection, and index.js calls
    31	    // startServer() -> db.LicenseFeedBacks.findAll() synchronously right after
    32	    // require('./models/db') returns. Mutate (Object.assign) rather than
    33	    // reassign so the `module.exports = db` reference stays valid. Awaiting
    34	    // before this line would leave consumers holding an empty object.
    35	    Object.assign(db, require('./db/init-models')(sequelize))
    36	    db.sequelize = sequelize
    37	
    38	    try {
    39	        await sequelize.authenticate();
    40	    } catch (error) {
    41	        console.error('Unable to connect to the database:', error);
    42	        throw 'Unable to connect to the database:', error
    43	    }
    44	
    45	    // Models are the schema source of truth: sync() creates brand-new tables
    46	    // (e.g. the sample-tracking tables) without altering existing ones.
    47	    await sequelize.sync();
    48	
    49	    // Idempotent, awaited migration block for FUTURE column additions. Runs
    50	    // AFTER sync() so it never races. Use snake_case identifiers only (mixed
    51	    // case folds to lowercase in unquoted Postgres DDL and orphans the column).
    52	    await applySampleMigrations(sequelize);
    53	}

    56	 * Additive, idempotent DDL for the sample-tracking module. Every statement is
    57	 * IF NOT EXISTS and individually guarded so a re-deploy never crashes boot.
    58	 */
    59	async function applySampleMigrations(sequelize) {
    60	    const schema = process.env.SCHEMA ? `"${process.env.SCHEMA}".` : '';
    61	    const statements = [
    62	        `ALTER TABLE ${schema}"sample" ADD COLUMN IF NOT EXISTS office_location VARCHAR(20)`,
    63	        `ALTER TABLE ${schema}"sample" ADD COLUMN IF NOT EXISTS next_stop VARCHAR(50)`,
    64	        `ALTER TABLE ${schema}"sample" ADD COLUMN IF NOT EXISTS final_destination VARCHAR(50)`,
    65	        `ALTER TABLE ${schema}"sample" ADD COLUMN IF NOT EXISTS notes TEXT`,
    66	        `ALTER TABLE ${schema}"sample" ADD COLUMN IF NOT EXISTS item_id_fk INTEGER`,
    67	        `ALTER TABLE ${schema}"sample" ADD COLUMN IF NOT EXISTS prod_order_no_fk VARCHAR(255)`,
    68	        `ALTER TABLE ${schema}"sample_box" ADD COLUMN IF NOT EXISTS final_destination VARCHAR(50)`,
    69	        `ALTER TABLE ${schema}"sample_event" ADD COLUMN IF NOT EXISTS notes TEXT`
    70	    ];
    71	    for (const stmt of statements) {
    72	        try {
    73	            await sequelize.query(stmt);
    74	        } catch (err) {
    75	            console.warn('Sample migration skipped:', err?.message);
    76	        }
    77	    }
    78	}

    26	// Translate one AG Grid filter definition into a Sequelize condition.
    27	// Mirrors the buildCondition logic used in lic.model.js.
    28	const buildCondition = (filterDef) => {
    29	  if (!filterDef) return null;
    30	
    31	  if (filterDef.filterType === 'date') {
    32	    const { type, dateFrom, dateTo } = filterDef;
    33	    if (!dateFrom) return null;
    34	    const startOfDay = new Date(dateFrom);
    35	    startOfDay.setHours(0, 0, 0, 0);
    36	    const endOfDay = new Date(startOfDay);
    37	    endOfDay.setHours(23, 59, 59, 999);
    38	    switch (type) {
    39	      case 'equals': return { [Op.between]: [startOfDay, endOfDay] };
    40	      case 'notEqual': return { [Op.notBetween]: [startOfDay, endOfDay] };
    41	      case 'lessThan': return { [Op.lt]: startOfDay };
    42	      case 'greaterThan': return { [Op.gt]: startOfDay };
    43	      case 'inRange': {
    44	        if (!dateTo) return null;
    45	        const rangeEnd = new Date(dateTo);
    46	        rangeEnd.setHours(23, 59, 59, 999);
    47	        return { [Op.between]: [startOfDay, rangeEnd] };
    48	      }
    49	      default: return null;
    50	    }
    51	  }
    52	
    53	  if (filterDef.filterType === 'number') {
    54	    const n = filterDef.filter;
    55	    switch (filterDef.type) {
    56	      case 'equals': return { [Op.eq]: n };
    57	      case 'notEqual': return { [Op.ne]: n };
    58	      case 'lessThan': return { [Op.lt]: n };
    59	      case 'lessThanOrEqual': return { [Op.lte]: n };
    60	      case 'greaterThan': return { [Op.gt]: n };
    61	      case 'greaterThanOrEqual': return { [Op.gte]: n };
    62	      case 'inRange': return { [Op.between]: [n, filterDef.filterTo] };
    63	      default: return null;
    64	    }
    65	  }
    66	
    67	  // default: text
    68	  const value = typeof filterDef.filter === 'string' ? filterDef.filter : '';
    69	  switch (filterDef.type) {
    70	    case 'contains': return { [Op.iLike]: `%${value}%` };
    71	    case 'notContains': return { [Op.notILike]: `%${value}%` };
    72	    case 'equals': return { [Op.eq]: filterDef.filter };
    73	    case 'notEqual': return { [Op.ne]: filterDef.filter };
    74	    case 'startsWith': return { [Op.iLike]: `${value}%` };
    75	    case 'endsWith': return { [Op.iLike]: `%${value}` };
    76	    case 'blank': return { [Op.or]: [{ [Op.eq]: null }, { [Op.eq]: '' }] };
    77	    case 'notBlank': return { [Op.and]: [{ [Op.ne]: null }, { [Op.ne]: '' }] };
    78	    default: return null;
    79	  }

    83	Sample.list = async (req, result) => {
    84	  const { startRow, endRow, filterModel } = req.query;
    85	  const limit = endRow != null && startRow != null ? Number(endRow) - Number(startRow) : 100;
    86	  const offset = startRow != null ? Number(startRow) : 0;
    87	
    88	  try {
    89	    const user = await getUserInfo(req.headers.authorization);
    90	    const filters = filterModel ? JSON.parse(filterModel) : {};
    91	    const whereClause = {};
    92	
    93	    Object.entries(filters).forEach(([field, filterDef]) => {
    94	      if (!FILTERABLE.has(field)) return;
    95	      // Normalise AG Grid "multi" filters the same way lic.model does.
    96	      let def = filterDef;
    97	      if (def && def.filterType === 'multi') {
    98	        def = (def.filterModels || []).find(sub => sub) || null;
    99	      }
   100	      const condition = buildCondition(def);
   101	      if (condition) whereClause[field] = condition;
   102	    });
   103	
   104	    if (user?.role === 'vendor') {
   105	      if (user.factory_id == null || Number.isNaN(Number(user.factory_id))) {
   106	        return result(null, { rows: [], totalCount: 0 });
   107	      }
   108	      whereClause.factory_id_fk = Number(user.factory_id);
   109	    }
   110	
   111	    const { count, rows } = await sql.sample.findAndCountAll({
   112	      where: whereClause,
   113	      order: [['sample_id_pk', 'DESC']],
   114	      limit,
   115	      offset,
   116	      distinct: true,
   117	      include: sql.sample_attachment
   118	        ? [{ model: sql.sample_attachment, as: 'sampleAttachments', required: false }]
   119	        : []
   120	    });
   121	
   122	    // Surface a primary_photo thumbnail URL on each row for the grid.
   123	    const mapped = rows.map(r => {
   124	      const plain = (r && typeof r.get === 'function') ? r.get({ plain: true }) : r;
   125	      const atts = (plain && plain.sampleAttachments) || [];
   126	      const primary = atts.find(a => a.primary_image) || atts[0];
   127	      plain.primary_photo = primary ? primary.attachment_link : null;
   128	      return plain;
   129	    });
   130	
   131	    // Surface the box label (so the grid shows which box a sample is in).
   132	    const boxIds = [...new Set(mapped.map(r => r.box_id_fk).filter(Boolean))];
   133	    if (boxIds.length && sql.sample_box) {
   134	      const boxes = await sql.sample_box.findAll({ where: { box_id_pk: { [Op.in]: boxIds } } });
   135	      const labels = new Map(boxes.map(b => [b.box_id_pk, b.box_label]));
   136	      mapped.forEach(r => {
   137	        r.box_label = r.box_id_fk ? (labels.get(r.box_id_fk) || `Box #${r.box_id_fk}`) : null;
   138	      });
   139	    } else {
   140	      mapped.forEach(r => { r.box_label = null; });
   141	    }
   142	
   143	    return result(null, { rows: mapped, totalCount: count });
   144	  } catch (err) {
   145	    console.error('Sample.list error', err);
   146	    return result(err);
   147	  }

   277	/**
   278	 * The bidirectional check-in/out engine. The client supplies event_type +
   279	 * office_location (+ optional notes, + to_status only for status_change). The
   280	 * resulting backbone status is DERIVED here and written under an optimistic
   281	 * guard so two concurrent moves from the same status cannot both succeed.
   282	 */
   283	Sample.recordEvent = async (req, result) => {
   284	  try {
   285	    const user = await getUserInfo(req.headers.authorization);
   286	    const body = req.body || {};
   287	    const sampleId = Number(body.sample_id_fk || req.params.id);
   288	    const eventType = body.event_type;
   289	    const office = body.office_location ?? null;
   290	    const notes = body.notes ?? null;
   291	    const requestedToStatus = body.to_status ?? null;
   292	
   293	    if (!sampleId) return result({ status: 400, message: 'sample_id_fk is required' });
   294	
   295	    const sample = await sql.sample.findByPk(sampleId);
   296	    if (!sample) return result({ status: 404, message: 'Sample not found' });
   297	
   298	    const fromStatus = sample.status;
   299	    const derived = status.deriveToStatus({
   300	      direction: sample.direction,
   301	      currentStatus: fromStatus,
   302	      office,
   303	      eventType,
   304	      requestedToStatus
   305	    });
   306	    if (!derived.ok) return result({ status: derived.status, message: derived.error });
   307	
   308	    // Optimistic concurrency: only advance if the row is still in fromStatus.
   309	    const [affected] = await sql.sample.update(
   310	      {
   311	        status: derived.toStatus,
   312	        office_location: office ?? sample.office_location,
   313	        sample_modUser: user?.name ?? null,
   314	        sample_modTime: String(Date.now())
   315	      },
   316	      { where: { sample_id_pk: sampleId, status: fromStatus } }
   317	    );
   318	    if (!affected) {
   319	      return result({ status: 409, message: 'Sample status changed concurrently; please retry' });
   320	    }
   321	
   322	    const event = await sql.sample_event.create({
   323	      sample_id_fk: sampleId,
   324	      event_type: eventType,
   325	      from_status: fromStatus,
   326	      to_status: derived.toStatus,
   327	      office_location: office,
   328	      event_user_id_fk: user?.user_id ?? null,
   329	      event_user: user?.name ?? null,
   330	      event_time: new Date(),
   331	      notes
   332	    });
   333	
   334	    return result(null, {
   335	      sample_id: sampleId,
   336	      from_status: fromStatus,
   337	      to_status: derived.toStatus,
   338	      event
   339	    });
   340	  } catch (err) {
   341	    console.error('Sample.recordEvent error', err);
   342	    return result(err);
   343	  }
   344	};

   185	/**
   186	 * Add samples to a box. A box can bundle samples from multiple factory groups.
   187	 * Direction must match (a sample cannot ride a box going the other way) — such
   188	 * samples are SKIPPED and reported, never blocking the rest (warn, don't block).
   189	 * Box status is independent: this does NOT change any sample's status.
   190	 */
   191	Box.addSamples = async (req, result) => {
   192	  try {
   193	    const user = await getUserInfo(req.headers.authorization);
   194	    const boxId = Number(req.params.id || req.body.box_id_fk);
   195	    const sampleIds = Array.isArray(req.body.sample_ids) ? req.body.sample_ids.map(Number) : [];
   196	    const legType = req.body.leg_type ?? null;
   197	
   198	    if (!boxId) return result({ status: 400, message: 'box id is required' });
   199	    if (!sampleIds.length) return result({ status: 400, message: 'sample_ids[] is required' });
   200	
   201	    const box = await sql.sample_box.findByPk(boxId);
   202	    if (!box) return result({ status: 404, message: 'Box not found' });
   203	
   204	    const samples = await sql.sample.findAll({ where: { sample_id_pk: { [Sequelize.Op.in]: sampleIds } } });
   205	    const found = new Map(samples.map(s => [s.sample_id_pk, s]));
   206	
   207	    const added = [];
   208	    const skipped = [];
   209	    for (const id of sampleIds) {
   210	      const s = found.get(id);
   211	      if (!s) { skipped.push({ sample_id: id, reason: 'not found' }); continue; }
   212	      if (box.direction && s.direction && box.direction !== s.direction) {
   213	        skipped.push({ sample_id: id, reason: `direction mismatch (sample ${s.direction} vs box ${box.direction})` });
   214	        continue;
   215	      }
   216	      await sql.sample_shipment_item.create({
   217	        sample_id_fk: id,
   218	        box_id_fk: boxId,
   219	        factory_group_id_fk: s.factory_group_id_fk ?? null,
   220	        leg_type: legType,
   221	        added_date: new Date(),
   222	        added_user: user?.name ?? null
   223	      });
   224	      await sql.sample.update({
   225	        box_id_fk: boxId,
   226	        next_stop: req.body.next_stop ?? box.dest_office ?? null,
   227	        final_destination: req.body.final_destination ?? box.final_destination ?? null
   228	      }, { where: { sample_id_pk: id } });
   229	      added.push(id);
   230	    }
   231	    return result(null, { box_id: boxId, added, skipped });
   232	  } catch (err) {
   233	    console.error('Box.addSamples error', err);
   234	    return result(err);
   235	  }
   236	};
