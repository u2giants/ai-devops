     1	/**
     2	 * Generic fuzzy-match of an input name against any array of records.
     3	 * nameField: the record property to match against (default 'licensor_name').
     4	 * Returns the matched record or null if no reasonable match found.
     5	 */
     6	function fuzzyMatch(inputName, records, nameField = 'licensor_name') {
     7	  if (!records || !records.length || !inputName) return null;
     8	  const input = (inputName || '').toLowerCase().trim();
     9	  const getName = r => (r[nameField] || '').toLowerCase().trim();
    10	
    11	  // 1. Exact match
    12	  let match = records.find(r => getName(r) === input);
    13	  if (match) return match;
    14	
    15	  // 2. Input contains the record name or vice versa
    16	  match = records.find(r => {
    17	    const name = getName(r);
    18	    return input.includes(name) || name.includes(input);
    19	  });
    20	  if (match) return match;
    21	
    22	  // 3. Word overlap scoring
    23	  const inputWords = input.split(/[\s\-\/\(\),]+/).filter(w => w.length > 2);
    24	  let bestScore = 0;
    25	  let bestMatch = null;
    26	  for (const r of records) {
    27	    const nameWords = getName(r).split(/[\s\-\/\(\),]+/).filter(w => w.length > 2);
    28	    let score = 0;
    29	    for (const word of inputWords) {
    30	      if (nameWords.some(nw => nw.includes(word) || word.includes(nw))) score++;
    31	    }
    32	    if (score > bestScore) {
    33	      bestScore = score;
    34	      bestMatch = r;
    35	    }
    36	  }
    37	  if (bestMatch && bestScore > 0) return bestMatch;
    38	
    39	  return null;
    40	}

    42	/** Backwards-compatible wrapper — matches against LicensingTime.licensor_name */
    43	function fuzzyMatchLicensor(inputName, records) {
    44	  return fuzzyMatch(inputName, records, 'licensor_name');
    45	}

    48	 * Add `days` calendar days to a date string or Date, returning a new Date.
    49	 * Returns null if the input date is invalid.
    50	 */
    51	function addDays(dateInput, days) {
    52	  if (!dateInput || days == null) return null;
    53	  const d = new Date(dateInput);
    54	  if (isNaN(d.getTime())) return null;
    55	  d.setDate(d.getDate() + Number(days));
    56	  return d;
    57	}

    61	 * Returns null for invalid dates.
    62	 */
    63	function formatDate(date) {
    64	  if (!date) return null;
    65	  const d = date instanceof Date ? date : new Date(date);
    66	  if (isNaN(d.getTime())) return null;
    67	  return `${d.getMonth() + 1}/${d.getDate()}/${d.getFullYear()}`;

     2	/**
     3	 * Sample status state machine — single source of truth for the sample lifecycle.
     4	 *
     5	 * A sample travels in one of two directions:
     6	 *   inbound  (factory-origin): requested -> shipped -> ningbo_checked_in
     7	 *                              -> ningbo_checked_out -> nyc_checked_in
     8	 *   outbound (usa-bought):     nyc_requested -> nyc_checked_out -> ningbo_checked_in
     9	 *                              -> ningbo_checked_out -> sent_to_factory
    10	 *
    11	 * Backbone moves (check_in / check_out / ship) have their resulting status
    12	 * DERIVED from (direction, current status, office, event_type) — the client
    13	 * never supplies the target status for those. The "shared extra" states
    14	 * (cancelled / on_hold / damaged / approved) are reached with a status_change
    15	 * event whose requested target is validated against a small allow-list.
    16	 */
    17

    19	const DIRECTIONS = ['inbound', 'outbound'];
    20	const OFFICES = ['Ningbo', 'NYC'];
    21	const EVENT_TYPES = ['check_in', 'check_out', 'ship', 'receive', 'status_change'];
    22	
    23	// Shared exception states reachable (almost) any time.
    24	const SHARED_EXTRAS = ['cancelled', 'on_hold', 'damaged', 'approved'];
    25	// Once here, no further moves are allowed.
    26	const TERMINAL_EXTRAS = ['cancelled'];

    28	// Origin -> how a freshly created sample starts.
    29	const ORIGIN_DEFAULTS = {
    30	  factory_request: { direction: 'inbound', initialStatus: 'requested' },
    31	  factory_offer: { direction: 'inbound', initialStatus: 'requested' },
    32	  usa_bought: { direction: 'outbound', initialStatus: 'nyc_requested' }

    64	function isBackboneStatus(direction, status) {
    65	  return (BACKBONE[direction] || []).includes(status);
    66	}

    68	/**
    69	 * Derive the resulting status for an event, or explain why it is rejected.
    70	 * @returns {{ ok: true, toStatus: string }
    71	 *          | { ok: false, status: number, error: string }}
    72	 */
    73	function deriveToStatus({ direction, currentStatus, office, eventType, requestedToStatus }) {
    74	  if (!DIRECTIONS.includes(direction)) {
    75	    return { ok: false, status: 400, error: `Unknown direction "${direction}"` };
    76	  }
    77	  if (!EVENT_TYPES.includes(eventType)) {
    78	    return { ok: false, status: 400, error: `Unknown event_type "${eventType}"` };
    79	  }
    80	  if (TERMINAL_EXTRAS.includes(currentStatus)) {
    81	    return { ok: false, status: 409, error: `Sample is ${currentStatus}; no further changes allowed` };
    82	  }
    83	
    84	  // Shared-extra / resume moves go through status_change.
    85	  if (eventType === 'status_change') {
    86	    if (!requestedToStatus) {
    87	      return { ok: false, status: 400, error: 'status_change requires a to_status' };
    88	    }
    89	    if (SHARED_EXTRAS.includes(requestedToStatus)) {
    90	      if (requestedToStatus === currentStatus) {
    91	        return { ok: false, status: 409, error: `Sample already ${currentStatus}` };
    92	      }
    93	      return { ok: true, toStatus: requestedToStatus };
    94	    }
    95	    // Resume from on_hold back onto the backbone.
    96	    if (currentStatus === 'on_hold' && isBackboneStatus(direction, requestedToStatus)) {
    97	      return { ok: true, toStatus: requestedToStatus };
    98	    }
    99	    return { ok: false, status: 400, error: `Illegal status_change to "${requestedToStatus}" from "${currentStatus}"` };
   100	  }
   101	
   102	  // Backbone moves: look up the single matching rule.
   103	  const rule = (TRANSITIONS[direction] || []).find(r =>
   104	    r.from === currentStatus &&
   105	    r.eventType === eventType &&
   106	    (r.office === null || r.office === office)
   107	  );
   108	  if (!rule) {
   109	    return {
   110	      ok: false,
   111	      status: 400,
   112	      error: `No ${eventType} transition from "${currentStatus}"` +
   113	        (office ? ` at ${office}` : '') + ` for ${direction} sample`
   114	    };
   115	  }
   116	  return { ok: true, toStatus: rule.to };
   117	}
